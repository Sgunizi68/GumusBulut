SELECT
    C.Alici_Unvani AS `e-Fatura Ünvan`,
    (SELECT K.Kategori_Adi
     FROM Kategori K
     WHERE K.Kategori_ID = C.e_Fatura_Kategori_ID) AS `e-Fatura Kategori`,
    CONCAT(
        ' #',
        O.Referans_ID,
        ' (',
        O.Referans_Metin,
        ' - ',
        (SELECT K2.Kategori_Adi
         FROM Kategori K2
         WHERE K2.Kategori_ID = O.Kategori_ID),
        ')'
    ) AS `Ödeme e-Referans`,
    C.Cari,
    C.Aktif_Pasif
FROM SilverCloud.Cari C
LEFT JOIN SilverCloud.Odeme_Referans O
    ON C.Referans_ID = O.Referans_ID;
