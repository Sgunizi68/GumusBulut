Önceki Dönem
SELECT sum(Tutar)
FROM SilverCloud.Gelir
WHERE
  Tarih < Ekran.Dönem.ilk_Gün
  AND Tarih >= Yemek_Ceki.İlk_Tarih
  AND Kategori_ID = Ekran.Kategori_ID

Sonraki Dönem

SELECT sum(Tutar)
FROM SilverCloud.Gelir
WHERE
  Tarih > Ekran.Dönem.Son_Gün
  AND Tarih <= Yemek_Ceki.Son_Tarih
  AND Kategori_ID = Ekran.Kategori_ID
