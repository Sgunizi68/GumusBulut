CASE
    WHEN EXISTS (
        SELECT 1
        FROM E_Fatura f
        INNER JOIN E_Fatura_Referans r
            ON f.Alici_Unvani = r.Alici_Unvani
        WHERE f.Giden_Fatura = 1
          AND f.Fatura_Tarihi = Ekran.SonTarih
          AND f.Tutar = Ekran.Tutar
          AND r.Referans_Kodu = Ekran.Kategori
    )
    THEN 'Kesildi'
    ELSE 'Beklemede'
END AS FaturaD
